    labelText: 'Public',
    state: 'on',
    size: 'md',
    hideLabel: false,
};

export const argTypes = Utils.hideStyledComponentProperties({
    state: {
        control: {
            type: 'select',
            options: ['on', 'off', 'status', 'disabled'],
        },
    },
    size: {

            type: 'select',
            options: ['sm', 'md', 'lg'],
        },
    },
    hideLabel: {

            type: 'radio',
            options: ['true', 'false'],
        },
    },
});

export const canvasExampleArgs = {
    ...defaultArgs,
};

export const CheckedCheckbox = (args) => {
    const [isChecked, setChecked] = useState(true);
    return <Checkbox {...args} onClick={() => setChecked(!isChecked)} checked={isChecked} />;
};
